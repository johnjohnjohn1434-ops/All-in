    <script>
        // Sample Initial Data Mocking Cloud uploads
        let mediaAssets = [
            { id: 1, type: 'video', title: 'Epic Mountain Nature', category: 'nature', file: 'https://assets.mixkit.co/videos/preview/mixkit-beautiful-aerial-view-of-a-mountain-valley-43033-large.mp4', thumb: 'https://images.unsplash.com/photo-1464822759023-fed622ff2c3b?w=500' },
            { id: 2, type: 'photo', title: 'Super Premium Car', category: 'cars', file: 'https://images.unsplash.com/photo-1503376780353-7e6692767b70?w=1200', thumb: 'https://images.unsplash.com/photo-1503376780353-7e6692767b70?w=500' },
            { id: 3, type: 'music', title: 'Aggressive Phonk Beat', category: 'phonk', file: 'https://www.soundhelix.com/examples/mp3/SoundHelix-Song-1.mp3', thumb: 'https://images.unsplash.com/photo-1614613535308-eb5fbd3d2c17?w=500' },
            { id: 4, type: 'sound', title: 'Heavy Thunder Sound', category: 'thunder', file: 'https://actions.google.com/sounds/v1/weather/rain_heavy_loud.ogg', thumb: 'https://images.unsplash.com/photo-1605721911519-3dfeb3be25e7?w=500' }
        ];

        const categoriesMap = {
            video: ['nature', 'gaming', 'cartoon', 'green screen', 'background'],
            photo: ['nature', 'cars', 'animals', 'wallpapers', 'ai images'],
            music: ['phonk', 'cinematic', 'funny', 'sad', 'motivation'],
            sound: ['explosion', 'rain', 'thunder', 'magic', 'horror']
        };

        let currentUser = null;
        let isAdmin = false;

        // Initialize App
        window.onload = function() {
            lucide.createIcons();
            renderGrid(mediaAssets, 'latest-grid');
            updateUploadCategories();
        };

        // Navigation Controller
        function showSection(section) {
            document.getElementById('home-section').classList.add('hidden');
            document.getElementById('category-section').classList.add('hidden');
            document.getElementById('admin-section').classList.add('hidden');

            if (section === 'home') {
                document.getElementById('home-section').classList.remove('hidden');
                renderGrid(mediaAssets, 'latest-grid');
            } else if (section === 'admin') {
                if (isAdmin) document.getElementById('admin-section').classList.remove('hidden');
            } else {
                let typeMap = { 'videos': 'video', 'photos': 'photo', 'music': 'music', 'sounds': 'sound' };
                let targetType = typeMap[section];
                
                document.getElementById('category-section').classList.remove('hidden');
                document.getElementById('category-title').innerText = section;
                document.getElementById('category-desc').innerText = `Explore our extreme library of ${section}`;
                
                // Render filter buttons
                let subTabs = document.getElementById('sub-categories');
                subTabs.innerHTML = `<button onclick="filterType('${targetType}')" class="bg-indigo-600 px-3 py-1.5 rounded-full text-xs font-semibold capitalize cursor-pointer">All</button>`;
                categoriesMap[targetType].forEach(cat => {
                    subTabs.innerHTML += `<button onclick="filterCategory('${targetType}', '${cat}')" class="bg-slate-800 hover:bg-slate-700 text-slate-300 px-3 py-1.5 rounded-full text-xs font-semibold capitalize cursor-pointer">${cat}</button>`;
                });

                filterType(targetType);
            }
        }

        // Render Cards
        function renderGrid(data, containerId) {
            let container = document.getElementById(containerId);
            container.innerHTML = '';

            if(data.length === 0) {
                container.innerHTML = `<p class="col-span-full text-center text-slate-500 py-12 text-sm">No assets found matching the criteria.</p>`;
                return;
            }

            data.forEach(item => {
                let previewElement = '';

                if (item.type === 'video') {
                    previewElement = `<video src="${item.file}" class="w-full h-44 object-cover" controls preload="none" poster="${item.thumb}"></video>`;
                } else if (item.type === 'photo') {
                    previewElement = `<img src="${item.thumb}" class="w-full h-44 object-cover cursor-pointer" onclick="window.open('${item.file}', '_blank')">`;
                } else if (item.type === 'music' || item.type === 'sound') {
                    previewElement = `
                        <div class="w-full h-44 bg-slate-900 flex flex-col justify-center items-center p-4 relative border-b border-slate-800">
                            <img src="${item.thumb}" class="absolute inset-0 w-full h-full object-cover opacity-20 pointer-events-none">
                            <i data-lucide="${item.type === 'music' ? 'music-4' : 'volume-2'}" class="w-10 h-10 text-indigo-400 z-10 mb-2"></i>
                            <audio src="${item.file}" controls class="w-full z-10 max-h-8 mt-2 opacity-80"></audio>
                        </div>
                    `;
                }

                let badgeColor = { video: 'bg-red-500/20 text-red-400', photo: 'bg-green-500/20 text-green-400', music: 'bg-blue-500/20 text-blue-400', sound: 'bg-amber-500/20 text-amber-400' };

                let cardHTML = `
                    <div class="bg-slate-900 border border-slate-800 rounded-xl overflow-hidden shadow-lg group hover:border-slate-700 transition">
                        <div class="relative">
                            ${previewElement}
                            <span class="absolute top-2 left-2 text-[10px] uppercase font-black tracking-widest px-2 py-0.5 rounded ${badgeColor[item.type]}">${item.type}</span>
                        </div>
                        <div class="p-4 space-y-3">
                            <div>
                                <h3 class="font-semibold text-sm truncate text-slate-200">${item.title}</h3>
                                <p class="text-slate-500 text-xs capitalize mt-0.5">${item.category}</p>
                            </div>
                            <a href="${item.file}" download="${item.title}" target="_blank" class="w-full bg-slate-800 hover:bg-indigo-600 transition text-slate-200 hover:text-white py-2 rounded-lg text-xs font-semibold flex items-center justify-center gap-1.5 cursor-pointer">
                                <i data-lucide="download" class="w-3.5 h-3.5"></i> Download Asset
                            </a>
                        </div>
                    </div>
                `;
                container.innerHTML += cardHTML;
            });
            lucide.createIcons();
        }

        // Filters
        function filterType(type) {
            let filtered = mediaAssets.filter(item => item.type === type);
            renderGrid(filtered, 'category-grid');
        }

        function filterCategory(type, category) {
            let filtered = mediaAssets.filter(item => item.type === type && item.category.toLowerCase() === category.toLowerCase());
            renderGrid(filtered, 'category-grid');
        }

        // Instant Live Search Bar
        function handleSearch(event) {
            let query = event.target.value.toLowerCase();
            let filtered = mediaAssets.filter(item => 
                item.title.toLowerCase().includes(query) || 
                item.category.toLowerCase().includes(query) ||
                item.type.toLowerCase().includes(query)
            );
            renderGrid(filtered, 'latest-grid');
        }

        // Authentication System (Secret Bypass)
        function openLoginModal() {
            document.getElementById('loginModal').classList.remove('hidden');
        }

        function closeLoginModal() {
            document.getElementById('loginModal').classList.add('hidden');
        }

        function processLogin() {
            let val = document.getElementById('loginInput').value.trim();
            if (!val) return;

            // Secret Admin Key Check
            if (val === "thivashthivash143@gmail.com*_*2009(*_*)/2009") {
                isAdmin = true;
                currentUser = "Admin";
                document.getElementById('userStatusBtn').innerText = "Dashboard";
                document.getElementById('loginBtn').onclick = () => showSection('admin');
                showSection('admin');
            } else {
                isAdmin = false;
                currentUser = val;
                document.getElementById('userStatusBtn').innerText = `Hi, ${val}`;
                alert(`Welcome User: ${val}`);
            }
            closeLoginModal();
        }

        function logoutAdmin() {
            isAdmin = false;
            currentUser = null;
            document.getElementById('userStatusBtn').innerText = "Login";
            document.getElementById('loginBtn').onclick = openLoginModal;
            showSection('home');
        }

        // Admin Upload Operations
        function updateUploadCategories() {
            let type = document.getElementById('uploadType').value;
            let catSelect = document.getElementById('uploadCategory');
            catSelect.innerHTML = '';
            categoriesMap[type].forEach(cat => {
                catSelect.innerHTML += `<option value="${cat}">${cat}</option>`;
            });
        }

        function handleUpload(event) {
            event.preventDefault();
            let type = document.getElementById('uploadType').value;
            let category = document.getElementById('uploadCategory').value;
            let title = document.getElementById('uploadTitle').value;
            let fileInput = document.getElementById('uploadFile');

            if(fileInput.files.length === 0) return;

            // Handle Direct Local File URL Creation
            let fileObjectURL = URL.createObjectURL(fileInput.files[0]);
            let defaultThumb = 'https://images.unsplash.com/photo-1618005182384-a83a8bd57fbe?w=500';

            let newAsset = {
                id: mediaAssets.length + 1,
                type: type,
                title: title,
                category: category,
                file: fileObjectURL,
                thumb: defaultThumb
            };

            // Insert into assets array at topmost
            mediaAssets.unshift(newAsset);
            alert("Asset published live successfully!");
            document.getElementById('uploadForm').reset();
            updateUploadCategories();
            showSection('home');
        }
    </script>
